const passwordInput = document.getElementById("password");
const strengthText = document.getElementById("strengthText");
const scoreText = document.getElementById("score");
const suggestionsList = document.getElementById("suggestionsList");
const strengthFill = document.getElementById("strengthFill");
const toggleBtn = document.getElementById("toggleBtn");

const commonPasswords = [
    "123456",
    "password",
    "admin",
    "qwerty",
    "welcome",
    "12345678"
];

passwordInput.addEventListener("input", analyzePassword);

function analyzePassword() {

    const password = passwordInput.value;
    let score = 0;
    let suggestions = [];

    if(password.length >= 8){
        score += 20;
    }else{
        suggestions.push("Use at least 8 characters.");
    }

    if(/[A-Z]/.test(password)){
        score += 20;
    }else{
        suggestions.push("Add an uppercase letter.");
    }

    if(/[a-z]/.test(password)){
        score += 20;
    }else{
        suggestions.push("Add a lowercase letter.");
    }

    if(/[0-9]/.test(password)){
        score += 20;
    }else{
        suggestions.push("Add a number.");
    }

    if(/[!@#$%^&*(),.?":{}|<>]/.test(password)){
        score += 20;
    }else{
        suggestions.push("Add a special character.");
    }

    if(commonPasswords.includes(password.toLowerCase())){
        score = 10;
        suggestions.push("This is a common password. Avoid using it.");
    }

    scoreText.innerText = `Score: ${score}/100`;

    if(score <= 40){
        strengthText.innerText = "Strength: Weak";
        strengthFill.style.width = "40%";
        strengthFill.style.background = "red";
    }
    else if(score <= 80){
        strengthText.innerText = "Strength: Medium";
        strengthFill.style.width = "70%";
        strengthFill.style.background = "orange";
    }
    else{
        strengthText.innerText = "Strength: Strong";
        strengthFill.style.width = "100%";
        strengthFill.style.background = "green";
    }

    suggestionsList.innerHTML = "";

    if(password.length === 0){
        strengthText.innerText = "Strength: -";
        scoreText.innerText = "Score: 0/100";
        strengthFill.style.width = "0%";
        return;
    }

    suggestions.forEach(item => {
        const li = document.createElement("li");
        li.textContent = item;
        suggestionsList.appendChild(li);
    });
}

toggleBtn.addEventListener("click", () => {

    if(passwordInput.type === "password"){
        passwordInput.type = "text";
        toggleBtn.innerText = "🙈";
    }
    else{
        passwordInput.type = "password";
        toggleBtn.innerText = "👁";
    }

});
